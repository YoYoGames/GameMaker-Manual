intervalo

A declaração break é utilizada para terminar prematuramente um for, repeat, while ou do / until loop de algum tipo, ou para dizer a um switch declaração para terminar nesse ponto, ou para terminar prematuramente um with chamada. Consulte as páginas individuais destas diferentes funções para obter uma explicação mais aprofundada de como ela pode ser usada em cada circunstância. Observe que se break for usado fora de qualquer um destes contextos, isso dará um erro.

Abaixo você pode ver alguns exemplos básicos de como break pode ser usado, e sua sintaxe é simples:

break;

break em um loop for:

for (var i = 0; i < 10; i += 1)
{
    if (array[i] == 234)
    {
        break;
    }
}

num = i;

break em um loop repeat:

var i = 0;
var temp = 0;
repeat (10)
{
    temp += array[i];
    if (temp > max_total)
    {
        break;
    }
    else
    {
        i += 1;
    }
}

break em um loop while:

var i = 0;
while (!place_free(x, y))
{
    x = random(room_width);
    y = random(room_height);
    if (i > 50)
    {
        break;
    }
    else
    {
        i+=1;
    }
}

break em um loop do / until:

var _id = noone;
do
{
    _id = list[| 0];
    if instance_exists(_id)
    {
        break;
    }
    ds_list_delete(list, 0);
}
until (ds_list_empty(list));
target = _id;

break ao utilizar with:

var count = 0;
with (obj_Enemy)
{
    count++;
    if (count > 10)
    {
        break;
    }
    hp = 100;
}

break em um switch:

switch (keyboard_key)
{
    case vk_left:
    case ord("A"):
        x -= 4;
    break;

    case vk_right:
    case ord("D"):
        x += 4;
    break;

    case vk_up:
    case ord("W"):
        y -= 4;
    break;

    case vk_down:
    case ord("S"):
        y += 4;
    break;
}
